### Example
```Makefile
.PHONY = test clean

SRCS := $(wildcard *.cpp)
objects := $(SRCS:%.cpp=%.o)

main: ${objects}
	@echo "Linking..."
	g++ $^ -o $@

%.o: %.cpp
	@echo compile $^ ...
	g++ -c $^ -o $@

test:
	#watch variable
	@echo ${SRCS}
	@echo ${BINS}

clean:
	@echo "Cleaning..."
	rm -rf *.o main


#　　$^    代表所有的依赖文件
#　　$@  代表所有的目标文件  
#　　$<   代表第一个依赖文件 
```
